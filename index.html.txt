import React, { useState, useEffect } from 'react';
import { Home, FileText, Users, Bell, Plus, X, Trash2, Upload, Image, Truck, Package, Calendar, Shield, BookOpen } from 'lucide-react';

export default function IntranetLaClave() {
  const [currentSection, setCurrentSection] = useState('inicio');
  const [posts, setPosts] = useState([]);
  const [policies, setPolicies] = useState([]);
  const [showNewPost, setShowNewPost] = useState(false);
  const [showNewPolicy, setShowNewPolicy] = useState(false);
  const [loading, setLoading] = useState(true);
  const [userName, setUserName] = useState('');
  const [isAdmin, setIsAdmin] = useState(false);
  const [showLogin, setShowLogin] = useState(true);
  const [showViewers, setShowViewers] = useState(null);
  const [newPost, setNewPost] = useState({
    title: '',
    content: '',
    category: 'noticias',
    image: null,
    files: []
  });
  const [newPolicy, setNewPolicy] = useState({
    title: '',
    content: '',
    type: 'manual',
    files: []
  });

  useEffect(() => {
    if (userName) {
      loadPosts();
      loadPolicies();
    }
  }, [userName]);

  const handleLogin = (name, adminStatus) => {
    setUserName(name);
    setIsAdmin(adminStatus);
    setShowLogin(false);
  };

  const handleLogout = () => {
    setUserName('');
    setIsAdmin(false);
    setShowLogin(true);
    setPosts([]);
  };

  const loadPosts = async () => {
    try {
      const result = await window.storage.list('post:', true);
      if (result && result.keys && result.keys.length > 0) {
        const loadedPosts = [];
        
        for (const key of result.keys) {
          try {
            const data = await window.storage.get(key, true);
            if (data && data.value) {
              const post = JSON.parse(data.value);
              loadedPosts.push(post);
              await markPostAsViewed(post.id);
            }
          } catch (err) {
            console.log('Error loading post:', key, err);
          }
        }
        
        setPosts(loadedPosts.sort((a, b) => b.timestamp - a.timestamp));
      } else {
        setPosts([]);
      }
    } catch (error) {
      console.log('No hay posts guardados a√∫n:', error);
      setPosts([]);
    }
    setLoading(false);
  };

  const loadPolicies = async () => {
    try {
      const result = await window.storage.list('policy:', true);
      if (result && result.keys && result.keys.length > 0) {
        const loadedPolicies = [];
        
        for (const key of result.keys) {
          try {
            const data = await window.storage.get(key, true);
            if (data && data.value) {
              const policy = JSON.parse(data.value);
              loadedPolicies.push(policy);
            }
          } catch (err) {
            console.log('Error loading policy:', key, err);
          }
        }
        
        setPolicies(loadedPolicies.sort((a, b) => a.type.localeCompare(b.type)));
      } else {
        setPolicies([]);
      }
    } catch (error) {
      console.log('No hay pol√≠ticas guardadas a√∫n:', error);
      setPolicies([]);
    }
  };

  const markPostAsViewed = async (postId) => {
    if (!userName) return;
    
    const viewKey = `view:${postId}:${userName}`;
    try {
      await window.storage.get(viewKey, true);
    } catch (error) {
      try {
        await window.storage.set(viewKey, JSON.stringify({
          userName,
          timestamp: Date.now(),
          date: new Date().toLocaleString('es-ES')
        }), true);
      } catch (e) {
        console.log('Error al marcar como visto:', e);
      }
    }
  };

  const getPostViewers = async (postId) => {
    try {
      const result = await window.storage.list(`view:${postId}:`, true);
      if (result && result.keys && result.keys.length > 0) {
        const viewers = [];
        
        for (const key of result.keys) {
          try {
            const data = await window.storage.get(key, true);
            if (data && data.value) {
              viewers.push(JSON.parse(data.value));
            }
          } catch (err) {
            console.log('Error loading viewer:', err);
          }
        }
        
        return viewers;
      }
      return [];
    } catch (error) {
      console.log('Error getting viewers:', error);
      return [];
    }
  };

  const showViewersModal = async (postId) => {
    const viewers = await getPostViewers(postId);
    setShowViewers({ postId, viewers });
  };

  const handleImageUpload = (e) => {
    const file = e.target.files[0];
    if (file && file.type.startsWith('image/')) {
      const reader = new FileReader();
      reader.onloadend = () => {
        setNewPost({ ...newPost, image: reader.result });
      };
      reader.readAsDataURL(file);
    }
  };

  const handleFileUpload = (e) => {
    const files = Array.from(e.target.files);
    const filePromises = files.map(file => {
      return new Promise((resolve) => {
        const reader = new FileReader();
        reader.onloadend = () => {
          resolve({
            name: file.name,
            type: file.type,
            data: reader.result
          });
        };
        reader.readAsDataURL(file);
      });
    });

    Promise.all(filePromises).then(fileData => {
      setNewPost({ ...newPost, files: [...newPost.files, ...fileData] });
    });
  };

  const removeFile = (index) => {
    const updatedFiles = newPost.files.filter((_, i) => i !== index);
    setNewPost({ ...newPost, files: updatedFiles });
  };

  const handleCreatePost = async () => {
    if (!newPost.title || !newPost.content) {
      alert('Por favor completa t√≠tulo y contenido');
      return;
    }

    const post = {
      id: Date.now().toString(),
      ...newPost,
      timestamp: Date.now(),
      date: new Date().toLocaleDateString('es-ES'),
      author: userName
    };

    try {
      await window.storage.set(`post:${post.id}`, JSON.stringify(post), true);
      setPosts([post, ...posts]);
      setNewPost({ title: '', content: '', category: 'noticias', image: null, files: [] });
      setShowNewPost(false);
      alert('Publicaci√≥n creada exitosamente');
    } catch (error) {
      alert('Error al guardar la publicaci√≥n: ' + error.message);
      console.error('Error:', error);
    }
  };

  const handleCreatePolicy = async () => {
    if (!newPolicy.title || !newPolicy.content) {
      alert('Por favor completa t√≠tulo y contenido');
      return;
    }

    const policy = {
      id: Date.now().toString(),
      ...newPolicy,
      timestamp: Date.now(),
      date: new Date().toLocaleDateString('es-ES'),
      author: userName
    };

    try {
      await window.storage.set(`policy:${policy.id}`, JSON.stringify(policy), true);
      setPolicies([...policies, policy].sort((a, b) => a.type.localeCompare(b.type)));
      setNewPolicy({ title: '', content: '', type: 'manual', files: [] });
      setShowNewPolicy(false);
      alert('Pol√≠tica creada exitosamente');
    } catch (error) {
      alert('Error al guardar la pol√≠tica: ' + error.message);
      console.error('Error:', error);
    }
  };

  const handleDeletePost = async (postId) => {
    if (!window.confirm('¬øEst√°s seguro de eliminar esta publicaci√≥n?')) {
      return;
    }

    try {
      await window.storage.delete(`post:${postId}`, true);
      setPosts(posts.filter(p => p.id !== postId));
      alert('Publicaci√≥n eliminada exitosamente');
    } catch (error) {
      console.error('Error al eliminar:', error);
      alert('Error al eliminar la publicaci√≥n: ' + error.message);
    }
  };

  const handleDeletePolicy = async (policyId) => {
    if (!window.confirm('¬øEst√°s seguro de eliminar esta pol√≠tica? Esta acci√≥n no se puede deshacer.')) {
      return;
    }

    try {
      await window.storage.delete(`policy:${policyId}`, true);
      setPolicies(policies.filter(p => p.id !== policyId));
      alert('Pol√≠tica eliminada exitosamente');
    } catch (error) {
      console.error('Error al eliminar:', error);
      alert('Error al eliminar la pol√≠tica: ' + error.message);
    }
  };

  const handlePolicyFileUpload = (e) => {
    const files = Array.from(e.target.files);
    const filePromises = files.map(file => {
      return new Promise((resolve) => {
        const reader = new FileReader();
        reader.onloadend = () => {
          resolve({
            name: file.name,
            type: file.type,
            data: reader.result
          });
        };
        reader.readAsDataURL(file);
      });
    });

    Promise.all(filePromises).then(fileData => {
      setNewPolicy({ ...newPolicy, files: [...newPolicy.files, ...fileData] });
    });
  };

  const removePolicyFile = (index) => {
    const updatedFiles = newPolicy.files.filter((_, i) => i !== index);
    setNewPolicy({ ...newPolicy, files: updatedFiles });
  };

  const filteredPosts = currentSection === 'inicio' 
    ? posts 
    : posts.filter(p => p.category === currentSection);

  if (showLogin) {
    return <LoginModal onLogin={handleLogin} />;
  }

  return (
    <div className="min-h-screen bg-gradient-to-br from-slate-100 via-blue-50 to-green-50">
      {/* Header con dise√±o mejorado */}
      <nav className="bg-white shadow-lg border-b-4 border-gradient-to-r from-blue-500 to-green-500">
        <div className="max-w-7xl mx-auto px-4 py-4">
          <div className="flex items-center justify-between">
            <div className="flex items-center gap-4">
              <div className="bg-gradient-to-br from-blue-500 to-green-500 p-3 rounded-xl shadow-lg">
                <Truck className="w-8 h-8 text-white" />
              </div>
              <div>
                <h1 className="text-3xl font-black text-transparent bg-clip-text bg-gradient-to-r from-blue-600 to-green-600">
                  LA CLAVE
                </h1>
                <p className="text-sm text-gray-600 font-semibold">ACCIONES SIN L√çMITES</p>
              </div>
            </div>
            
            <div className="flex items-center gap-6">
              <div className="flex items-center gap-3">
                <Package className="w-5 h-5 text-blue-500" />
                <Calendar className="w-5 h-5 text-green-500" />
              </div>
              <div className="text-right">
                <p className="text-sm text-gray-600">Bienvenido,</p>
                <p className="font-bold text-gray-800 text-lg">{userName}</p>
                {isAdmin && (
                  <span className="inline-block text-xs bg-gradient-to-r from-blue-500 to-green-500 text-white px-3 py-1 rounded-full font-semibold shadow-md">
                    Administrador
                  </span>
                )}
              </div>
              <button
                onClick={handleLogout}
                className="px-6 py-2 text-sm font-semibold text-white bg-gradient-to-r from-red-500 to-red-600 hover:from-red-600 hover:to-red-700 rounded-xl shadow-lg transition-all transform hover:scale-105"
              >
                Salir
              </button>
            </div>
          </div>
        </div>
      </nav>

      <div className="max-w-7xl mx-auto px-4 py-8 flex gap-6">
        {/* Sidebar mejorado */}
        <aside className="w-72 h-fit sticky top-8">
          <div className="bg-white rounded-2xl shadow-xl p-6 border border-gray-100">
            <div className="mb-6 pb-4 border-b border-gray-200">
              <h3 className="text-sm font-bold text-gray-500 uppercase tracking-wider mb-1">Navegaci√≥n</h3>
              <p className="text-xs text-gray-400">Panel de Control</p>
            </div>
            <nav className="space-y-3">
              <button
                onClick={() => setCurrentSection('inicio')}
                className={`w-full flex items-center gap-4 px-5 py-4 rounded-xl transition-all transform hover:scale-105 ${
                  currentSection === 'inicio' 
                    ? 'bg-gradient-to-r from-blue-500 to-green-500 text-white shadow-lg' 
                    : 'text-gray-700 hover:bg-gray-50 border border-gray-200'
                }`}
              >
                <Home className="w-6 h-6" />
                <span className="font-bold">Inicio</span>
              </button>
              <button
                onClick={() => setCurrentSection('politicas')}
                className={`w-full flex items-center gap-4 px-5 py-4 rounded-xl transition-all transform hover:scale-105 ${
                  currentSection === 'politicas' 
                    ? 'bg-gradient-to-r from-blue-500 to-green-500 text-white shadow-lg' 
                    : 'text-gray-700 hover:bg-gray-50 border border-gray-200'
                }`}
              >
                <Shield className="w-6 h-6" />
                <span className="font-bold">Pol√≠ticas</span>
              </button>
              <button
                onClick={() => setCurrentSection('noticias')}
                className={`w-full flex items-center gap-4 px-5 py-4 rounded-xl transition-all transform hover:scale-105 ${
                  currentSection === 'noticias' 
                    ? 'bg-gradient-to-r from-blue-500 to-green-500 text-white shadow-lg' 
                    : 'text-gray-700 hover:bg-gray-50 border border-gray-200'
                }`}
              >
                <Bell className="w-6 h-6" />
                <span className="font-bold">Noticias</span>
              </button>
              <button
                onClick={() => setCurrentSection('documentos')}
                className={`w-full flex items-center gap-4 px-5 py-4 rounded-xl transition-all transform hover:scale-105 ${
                  currentSection === 'documentos' 
                    ? 'bg-gradient-to-r from-blue-500 to-green-500 text-white shadow-lg' 
                    : 'text-gray-700 hover:bg-gray-50 border border-gray-200'
                }`}
              >
                <FileText className="w-6 h-6" />
                <span className="font-bold">Documentos</span>
              </button>
              <button
                onClick={() => setCurrentSection('equipo')}
                className={`w-full flex items-center gap-4 px-5 py-4 rounded-xl transition-all transform hover:scale-105 ${
                  currentSection === 'equipo' 
                    ? 'bg-gradient-to-r from-blue-500 to-green-500 text-white shadow-lg' 
                    : 'text-gray-700 hover:bg-gray-50 border border-gray-200'
                }`}
              >
                <Users className="w-6 h-6" />
                <span className="font-bold">Equipo</span>
              </button>
            </nav>
          </div>

          {/* Stats card */}
          <div className="mt-6 bg-gradient-to-br from-blue-500 to-green-500 rounded-2xl shadow-xl p-6 text-white">
            <h4 className="font-bold text-lg mb-2">Total Publicaciones</h4>
            <p className="text-4xl font-black">{posts.length}</p>
            <p className="text-sm opacity-90 mt-2">Actualizaciones del equipo</p>
          </div>

          <div className="mt-4 bg-gradient-to-br from-purple-500 to-blue-500 rounded-2xl shadow-xl p-6 text-white">
            <h4 className="font-bold text-lg mb-2">Pol√≠ticas Activas</h4>
            <p className="text-4xl font-black">{policies.length}</p>
            <p className="text-sm opacity-90 mt-2">Documentos oficiales</p>
          </div>
        </aside>

        <main className="flex-1">
          {/* Header section mejorado */}
          <div className="bg-white rounded-2xl shadow-xl p-8 mb-6 border border-gray-100">
            <div className="flex items-center justify-between">
              <div>
                <h2 className="text-3xl font-black text-transparent bg-clip-text bg-gradient-to-r from-blue-600 to-green-600 mb-2">
                  {currentSection === 'inicio' && 'Todas las Publicaciones'}
                  {currentSection === 'politicas' && 'Pol√≠ticas y Normativas'}
                  {currentSection === 'noticias' && 'Noticias'}
                  {currentSection === 'documentos' && 'Documentos'}
                  {currentSection === 'equipo' && 'Equipo'}
                </h2>
                <p className="text-gray-600">
                  {currentSection === 'politicas' 
                    ? 'Documentos oficiales, manuales y protocolos de la empresa'
                    : 'Mantente al d√≠a con las √∫ltimas actualizaciones'}
                </p>
              </div>
              {isAdmin && (
                <div className="flex gap-3">
                  {currentSection === 'politicas' && (
                    <button
                      onClick={() => setShowNewPolicy(true)}
                      className="flex items-center gap-3 bg-gradient-to-r from-purple-500 to-blue-500 text-white px-6 py-4 rounded-xl hover:shadow-2xl transition-all transform hover:scale-105 font-bold"
                    >
                      <Shield className="w-6 h-6" />
                      Nueva Pol√≠tica
                    </button>
                  )}
                  {currentSection !== 'politicas' && (
                    <button
                      onClick={() => setShowNewPost(true)}
                      className="flex items-center gap-3 bg-gradient-to-r from-blue-500 to-green-500 text-white px-6 py-4 rounded-xl hover:shadow-2xl transition-all transform hover:scale-105 font-bold"
                    >
                      <Plus className="w-6 h-6" />
                      Nueva Publicaci√≥n
                    </button>
                  )}
                </div>
              )}
            </div>
          </div>

          {showNewPost && (
            <div className="fixed inset-0 bg-black bg-opacity-60 backdrop-blur-sm flex items-center justify-center p-4 z-50 overflow-y-auto">
              <div className="bg-white rounded-2xl shadow-2xl max-w-2xl w-full p-8 my-8 border border-gray-200">
                <div className="flex items-center justify-between mb-6">
                  <h3 className="text-2xl font-black text-transparent bg-clip-text bg-gradient-to-r from-blue-600 to-green-600">
                    Nueva Publicaci√≥n
                  </h3>
                  <button
                    onClick={() => setShowNewPost(false)}
                    className="text-gray-500 hover:text-gray-700 hover:bg-gray-100 p-2 rounded-lg transition-colors"
                  >
                    <X className="w-6 h-6" />
                  </button>
                </div>
                <div className="space-y-5">
                  <div>
                    <label className="block text-sm font-bold text-gray-700 mb-2">T√≠tulo</label>
                    <input
                      type="text"
                      value={newPost.title}
                      onChange={(e) => setNewPost({ ...newPost, title: e.target.value })}
                      className="w-full px-4 py-3 border-2 border-gray-300 rounded-xl focus:ring-2 focus:ring-blue-500 focus:border-transparent transition-all"
                      placeholder="T√≠tulo de la publicaci√≥n"
                    />
                  </div>
                  <div>
                    <label className="block text-sm font-bold text-gray-700 mb-2">Categor√≠a</label>
                    <select
                      value={newPost.category}
                      onChange={(e) => setNewPost({ ...newPost, category: e.target.value })}
                      className="w-full px-4 py-3 border-2 border-gray-300 rounded-xl focus:ring-2 focus:ring-blue-500 focus:border-transparent transition-all"
                    >
                      <option value="noticias">Noticias</option>
                      <option value="documentos">Documentos</option>
                      <option value="equipo">Equipo</option>
                    </select>
                  </div>
                  <div>
                    <label className="block text-sm font-bold text-gray-700 mb-2">Contenido</label>
                    <textarea
                      value={newPost.content}
                      onChange={(e) => setNewPost({ ...newPost, content: e.target.value })}
                      className="w-full px-4 py-3 border-2 border-gray-300 rounded-xl focus:ring-2 focus:ring-blue-500 focus:border-transparent h-32 transition-all"
                      placeholder="Escribe el contenido aqu√≠..."
                    />
                  </div>
                  
                  <div>
                    <label className="block text-sm font-bold text-gray-700 mb-2">Imagen</label>
                    <div className="flex items-center gap-2">
                      <label className="flex items-center gap-2 px-5 py-3 bg-blue-50 text-blue-600 rounded-xl hover:bg-blue-100 cursor-pointer transition-all font-semibold border-2 border-blue-200">
                        <Image className="w-5 h-5" />
                        <span>Subir Imagen</span>
                        <input
                          type="file"
                          accept="image/*"
                          onChange={handleImageUpload}
                          className="hidden"
                        />
                      </label>
                      {newPost.image && (
                        <button
                          onClick={() => setNewPost({ ...newPost, image: null })}
                          className="px-4 py-3 text-red-600 hover:bg-red-50 rounded-xl font-semibold border-2 border-red-200 transition-all"
                        >
                          Quitar
                        </button>
                      )}
                    </div>
                    {newPost.image && (
                      <img src={newPost.image} alt="Preview" className="mt-3 max-h-48 rounded-xl border-4 border-gray-200 shadow-lg" />
                    )}
                  </div>

                  <div>
                    <label className="block text-sm font-bold text-gray-700 mb-2">Archivos Adjuntos</label>
                    <label className="flex items-center gap-2 px-5 py-3 bg-green-50 text-green-600 rounded-xl hover:bg-green-100 cursor-pointer transition-all w-fit font-semibold border-2 border-green-200">
                      <Upload className="w-5 h-5" />
                      <span>Subir Archivos</span>
                      <input
                        type="file"
                        multiple
                        onChange={handleFileUpload}
                        className="hidden"
                      />
                    </label>
                    {newPost.files.length > 0 && (
                      <div className="mt-3 space-y-2">
                        {newPost.files.map((file, index) => (
                          <div key={index} className="flex items-center justify-between bg-gray-50 p-3 rounded-xl border-2 border-gray-200">

                              className="text-red-600 hover:text-red-700 p-1 hover:bg-red-50 rounded-lg transition-all"

                              <X className="w-5 h-5" />
                            </button>
                          </div>
                        ))}
                      </div>
                    )}
                  </div>

                  <div className="flex gap-3 pt-4">
                    <button
                      onClick={handleCreatePost}
                      className="flex-1 bg-gradient-to-r from-blue-500 to-green-500 text-white py-4 rounded-xl font-bold hover:shadow-xl transition-all transform hover:scale-105"
                    >
                      Publicar
                    </button>
                    <button
                      onClick={() => setShowNewPost(false)}
                      className="flex-1 bg-gray-200 text-gray-800 py-4 rounded-xl font-bold hover:bg-gray-300 transition-all"
                    >
                      Cancelar
                    </button>
                  </div>
                </div>
              </div>
            </div>

          {showNewPolicy && (
            <div className="fixed inset-0 bg-black bg-opacity-60 backdrop-blur-sm flex items-center justify-center p-4 z-50 overflow-y-auto">
              <div className="bg-white rounded-2xl shadow-2xl max-w-2xl w-full p-8 my-8 border border-gray-200">
                <div className="flex items-center justify-between mb-6">
                  <h3 className="text-2xl font-black text-transparent bg-clip-text bg-gradient-to-r from-purple-600 to-blue-600">
                    Nueva Pol√≠tica
                  </h3>
                  <button
                    onClick={() => setShowNewPolicy(false)}
                    className="text-gray-500 hover:text-gray-700 hover:bg-gray-100 p-2 rounded-lg transition-colors"
                  >
                    <X className="w-6 h-6" />
                  </button>
                </div>
                <div className="space-y-5">
                  <div>
                    <label className="block text-sm font-bold text-gray-700 mb-2">T√≠tulo</label>
                    <input
                      type="text"
                      value={newPolicy.title}
                      onChange={(e) => setNewPolicy({ ...newPolicy, title: e.target.value })}
                      className="w-full px-4 py-3 border-2 border-gray-300 rounded-xl focus:ring-2 focus:ring-purple-500 focus:border-transparent transition-all"
                      placeholder="Ej: Manual del Trabajador 2024"
                    />
                  </div>
                  <div>
                    <label className="block text-sm font-bold text-gray-700 mb-2">Tipo de Pol√≠tica</label>
                    <select
                      value={newPolicy.type}
                      onChange={(e) => setNewPolicy({ ...newPolicy, type: e.target.value })}
                      className="w-full px-4 py-3 border-2 border-gray-300 rounded-xl focus:ring-2 focus:ring-purple-500 focus:border-transparent transition-all"
                    >
                      <option value="manual">Manual del Trabajador</option>
                      <option value="reglamento">Reglamento Interno</option>
                      <option value="conducta">C√≥digo de Conducta</option>
                      <option value="seguridad">Protocolos de Seguridad</option>
                      <option value="privacidad">Pol√≠ticas de Privacidad</option>
                      <option value="otro">Otro</option>
                    </select>
                  </div>
                  <div>
